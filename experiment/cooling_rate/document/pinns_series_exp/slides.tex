\begin{frame}{Global Introduction}
    \begin{itemize}
        \item \textbf{Objective:} Systematic evaluation of Physics-Informed Neural Network (PINN) performance.
        \item \textbf{Scope:} 18 experiments testing the sensitivity to loss weights, collocation density, and data sparsity.
        \item \textbf{Methodology:} One-Factor-at-a-Time (OFAT) parameter variation.
        \item \textbf{Key Metric:} Percentage error of the discovered cooling rate $R$ against the true physical value.
    \end{itemize}
\end{frame}
